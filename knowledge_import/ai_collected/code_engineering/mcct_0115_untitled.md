19	# ---------------------------------------------------------------------------
20	# 1. Floyd-Warshall  —  O(V^3)
21	# ---------------------------------------------------------------------------
22	
23	def floyd_warshall(
24	    graph: list[list[float]], *, detect_negative_cycle: bool = False
25	) -> list[list[float]]:
26	    """Return shortest-distance matrix for all vertex pairs.
27	
28	    graph[i][j] = weight of edge i→j, or math.inf if no direct edge.
29	    Diagonal entries (self-loops) are forced to 0 so negative-cycle
30	    detection on the diagonal works correctly.
31	
32	    Raises ValueError on malformed input (empty or non-square matrix).
33	    If detect_negative_cycle is True, raises ValueError when a
34	    negative cycle is found (dist[i][i] < 0 for some i).
35	    """
36	    n = len(graph)
37	    if n == 0:
38	        raise ValueError("graph must have at least one vertex")
39	    for i, row in enumerate(graph):
40	        if len(row) != n:
41	            raise ValueError(f"row {i} has {len(row)} cols, expected {n}")
42	
43	    dist = [row[:] for row in graph]
44	    for i in range(n):
45	        dist[i][i] = 0
46	
47	    for k in range(n):
48	        dk = dist[k]  # cache row k for locality
49	        for i in range(n):
50	            dik = dist[i][k]
51	            if dik == INF:
52	                continue
53	            di = dist[i]
54	            for j in range(n):
55	                alt = dik + dk[j]
56	                if alt < di[j]:
57	                    di[j] = alt
58	
59	    if detect_negative_cycle and any(dist[i][i] < 0 for i in range(n)):
60	        raise ValueError("negative cycle detected")
61	
62	    return dist

65	def floyd_warshall_with_path(
66	    graph: list[list[float]],
67	) -> tuple[list[list[float]], list[list[Optional[int]]]]:
68	    """Like floyd_warshall, but also returns a successor matrix for path reconstruction."""
69	    n = len(graph)
70	    if n == 0:
71	        raise ValueError("graph must have at least one vertex")
72	    for i, row in enumerate(graph):
73	        if len(row) != n:
74	            raise ValueError(f"row {i} has {len(row)} cols, expected {n}")
75	
76	    dist = [row[:] for row in graph]
77	    for i in range(n):
78	        dist[i][i] = 0
79	
80	    nxt: list[list[Optional[int]]] = [[None] * n for _ in range(n)]
81	    for i in range(n):
82	        for j in range(n):
83	            if i != j and graph[i][j] != INF:
84	                nxt[i][j] = j
85	
86	    for k in range(n):
87	        dk = dist[k]
88	        for i in range(n):
89	            dik = dist[i][k]
90	            if dik == INF:
91	                continue
92	            di = dist[i]
93	            ni = nxt[i]
94	            nk = nxt[k]
95	            for j in range(n):
96	                alt = dik + dk[j]
97	                if alt < di[j]:
98	                    di[j] = alt
99	                    ni[j] = ni[k]
100	
101	    return dist, nxt

104	def reconstruct_path(nxt: list[list[int | None]], u: int, v: int) -> list[int] | None:
105	    """Reconstruct the shortest path from u to v using the successor matrix."""
106	    if nxt[u][v] is None:
107	        return None
108	    path = [u]
109	    while u != v:
110	        u = nxt[u][v]
111	        if u is None:
112	            return None
113	        path.append(u)
114	    return path

117	# ---------------------------------------------------------------------------
118	# 2. Brute-force: Bellman-Ford from every vertex  —  O(V * V * E)
119	# ---------------------------------------------------------------------------
120	
121	def bellman_ford(
122	    n: int, edges: list[tuple[int, int, float]], src: int
123	) -> Optional[list[float]]:
124	    """Single-source shortest paths via Bellman-Ford.
125	
126	    Returns None if a negative cycle is reachable from src.
127	    """
128	    dist = [INF] * n
129	    dist[src] = 0
130	
131	    for _ in range(n - 1):
132	        updated = False
133	        for u, v, w in edges:
134	            if dist[u] != INF and dist[u] + w < dist[v]:
135	                dist[v] = dist[u] + w
136	                updated = True
137	        if not updated:
138	            break
139	
140	    # negative-cycle check
141	    for u, v, w in edges:
142	        if dist[u] != INF and dist[u] + w < dist[v]:
143	            return None
144	
145	    return dist

148	def brute_force_apsp(graph: list[list[float]]) -> Optional[list[list[float]]]:
149	    """All-pairs shortest paths by running Bellman-Ford from every vertex."""
150	    n = len(graph)
151	    edges = [
152	        (i, j, graph[i][j])
153	        for i in range(n)
154	        for j in range(n)
155	        if i != j and graph[i][j] != INF
156	    ]
157	
158	    result: list[list[float]] = []
159	    for src in range(n):
160	        d = bellman_ford(n, edges, src)
161	        if d is None:
162	            return None
163	        result.append(d)
164	    return result
