1	"""Breadth-First Search (BFS) for shortest path in an unweighted graph."""
2	
3	from collections import deque
4	
5	
6	def bfs_shortest_path(graph: dict[int, list[int]], start: int, end: int) -> list[int] | None:
7	    """Find the shortest path between two nodes in an unweighted graph using BFS.
8	
9	    BFS explores all neighbors at the current depth before moving to the next
10	    level, which guarantees the first time we reach the target, we've found a
11	    path with the fewest edges.
12	
13	    Args:
14	        graph: Adjacency list mapping each node to its list of neighbors.
15	        start: The source node.
16	        end:   The destination node.
17	
18	    Returns:
19	        A list of nodes representing the shortest path from start to end
20	        (inclusive), or None if no path exists.
21	    """
22	    # A queue of paths discovered so far. Each entry is a full path from
23	    # `start` to the node at the front of that path.  We use a deque so
24	    # that popping from the left (FIFO) is O(1).
25	    queue: deque[list[int]] = deque([[start]])
26	
27	    # Keep track of visited nodes to avoid revisiting them (and getting
28	    # stuck in cycles).  A set gives O(1) membership checks.
29	    visited: set[int] = {start}
30	
31	    while queue:
32	        # Grab the oldest path — BFS processes nodes in order of their
33	        # distance from `start`, so the first path that reaches `end`
34	        # is guaranteed to be a shortest one.
35	        path = queue.popleft()
36	        node = path[-1]  # The node at the tip of the current path.
37	
38	        # Goal check: if we've reached the destination, return the path.
39	        if node == end:
40	            return path
41	
42	        # Expand: enqueue one new path for every unvisited neighbor.
43	        for neighbor in graph.get(node, []):
44	            if neighbor not in visited:
45	                visited.add(neighbor)
46	                # Build a new path by appending the neighbor.  Using
47	                # `path + [neighbor]` creates a fresh list so that
48	                # concurrent branches don't interfere with each other.
49	                queue.append(path + [neighbor])
50	
51	    # The queue emptied without finding `end` — no path exists.
52	    return None
53

1	"""Breadth-First Search (BFS) for shortest path in unweighted graphs.
2	
3	BFS explores a graph level by level — all neighbors of the current node
4	before moving to the next depth.  This guarantees that the first time a
5	vertex is reached, it is via a shortest path (fewest edges).
6	
7	Two main functions are provided:
8	  1. bfs_shortest_path — returns the shortest path between two vertices.
9	  2. bfs_shortest_distances — returns shortest distances from a source
10	     to every reachable vertex.
11	
12	Time complexity:  O(V + E)
13	Space complexity: O(V)
14	"""
15	
16	from collections import defaultdict, deque
17	
18	
19	class Graph:
20	    """Graph stored as an adjacency list.
21	
22	    Each vertex maps to a list of its neighbors.
23	    An optional `directed` flag controls whether edges are one-way or two-way.
24	    """
25	
26	    def __init__(self, directed=False):
27	        self.adj = defaultdict(list)
28	        self.directed = directed
29	
30	    def add_edge(self, u, v):
31	        """Add an edge from u to v.  For undirected graphs, also add v to u."""
32	        self.adj[u].append(v)
33	        if not self.directed:
34	            self.adj[v].append(u)
35	        else:
36	            # Ensure target vertex exists in the adjacency list even if it
37	            # has no outgoing edges — otherwise lookups will KeyError.
38	            self.adj[v]
39	
40	    def bfs_shortest_path(self, start, target):
41	        """Find the shortest path from start to target using BFS.
42	
43	        Args:
44	            start:  The source vertex.
45	            target: The destination vertex.
46	
47	        Returns:
48	            A tuple (path, length) where:
49	              - path is a list of vertices from start to target (inclusive),
50	                or an empty list if no path exists.
51	              - length is the number of edges in the path, or 0 if start
52	                equals target, or float("inf") if unreachable.
53	
54	        Raises:
55	            KeyError: If start or target is not in the graph.
56	        """
57	        if start not in self.adj:
58	            raise KeyError(f"source vertex {start!r} not in graph")
59	        if target not in self.adj:
60	            raise KeyError(f"target vertex {target!r} not in graph")
61	
62	        if start == target:
63	            return [start], 0
64	
65	        visited = {start}
66	        # Each queue entry is (vertex, path_from_start)
67	        queue = deque([(start, [start])])
68	
69	        while queue:
70	            node, path = queue.popleft()
71	
72	            for neighbor in self.adj[node]:
73	                if neighbor in visited:
74	                    continue
75	                new_path = path + [neighbor]
76	                if neighbor == target:
77	                    return new_path, len(new_path) - 1
78	                visited.add(neighbor)
79	                queue.append((neighbor, new_path))
80	
81	        # Queue exhausted — target not reachable
82	        return [], float("inf")
83	
84	    def bfs_shortest_distances(self, start):
85	        """Compute shortest distances from start to every reachable vertex.
86	
87	        Args:
88	            start: The source vertex.
89	
90	        Returns:
91	            A dict mapping each reachable vertex to its distance (number of
92	            edges) from start.  The start vertex maps to 0.
93	
94	        Raises:
95	            KeyError: If start is not in the graph.
96	        """
97	        if start not in self.adj:
98	            raise KeyError(f"source vertex {start!r} not in graph")
99	
100	        distances = {start: 0}
101	        queue = deque([start])
102	
103	        while queue:
104	            node = queue.popleft()
105	            for neighbor in self.adj[node]:
106	                if neighbor not in distances:
107	                    distances[neighbor] = distances[node] + 1
108	                    queue.append(neighbor)
109	
110	        return distances
111	
112	    def bfs_traversal(self, start):
113	        """Return all vertices reachable from start in BFS order.
114	
115	        Args:
116	            start: The source vertex.
117	
118	        Returns:
119	            A list of vertices in the order they were visited.
120	
121	        Raises:
122	            KeyError: If start is not in the graph.
123	        """
124	        if start not in self.adj:
125	            raise KeyError(f"source vertex {start!r} not in graph")
126	
127	        visited = [start]
128	        seen = {start}
129	        queue = deque([start])
130	
131	        while queue:
132	            node = queue.popleft()
133	            for neighbor in self.adj[node]:
134	                if neighbor not in seen:
135	                    seen.add(neighbor)
136	                    visited.append(neighbor)
137	                    queue.append(neighbor)
138	
139	        return visited
140

11	def _grid(width, height, walls=None):
12	    """Build a 4-connected grid graph with optional wall edges."""
13	    walls = walls or set()
14	    g = Graph()
15	    # Ensure all cells exist as vertices even if they end up isolated.
16	    for x in range(width):
17	        for y in range(height):
18	            g.adj[(x, y)]
19	    for x in range(width):
20	        for y in range(height):
21	            for dx, dy in [(1, 0), (0, 1)]:
22	                nx, ny = x + dx, y + dy
23	                if 0 <= nx < width and 0 <= ny < height:
24	                    edge = ((x, y), (nx, ny))
25	                    if edge not in walls:
