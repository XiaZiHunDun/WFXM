11	 * Information about a duplicate value found in the data.
12	 */
13	class DuplicateInfo {
14	  constructor(value, paths) {
15	    this.value = value;
16	    this.paths = paths;
17	  }
18	
19	  toString() {
20	    return `DuplicateInfo(value=${JSON.stringify(this.value)}, paths=${JSON.stringify(this.paths)})`;
21	  }
22	
23	  equals(other) {
24	    if (!(other instanceof DuplicateInfo)) return false;
25	    return (
26	      deepEqual(this.value, other.value) &&
27	      arraysEqual(this.paths.slice().sort(), other.paths.slice().sort())
28	    );
29	  }
30	}

32	function deepEqual(a, b) {
33	  return JSON.stringify(a, JSON.stringify(a, Object.keys(a || {}).sort())) ===
34	    JSON.stringify(b, JSON.stringify(b, Object.keys(b || {}).sort()));
35	}

37	function arraysEqual(a, b) {
38	  if (a.length !== b.length) return false;
39	  for (let i = 0; i < a.length; i++) {
40	    if (a[i] !== b[i]) return false;
41	  }
42	  return true;
43	}

46	 * Convert a value to a stable JSON string for comparison.

48	function toComparable(value) {
49	  try {
50	    return JSON.stringify(value, Object.keys(value || {}).sort());
51	  } catch {
52	    return String(value);

54	}

67	/**
68	 * Find duplicate values in data validated against a JSON schema.
69	 *
70	 * @param {*} data - The value to inspect (object, array, or primitive).
71	 * @param {Object} schema - The JSON schema definition for this value.
72	 * @param {string} [path="$"] - JSONPath-like location of the current value.
73	 * @returns {DuplicateInfo[]}
74	 */
75	function findDuplicates(data, schema, path) {
76	  if (path === undefined) path = "$";
77	  return findInValue(data, schema, path);
78	}
79	
80	function findInValue(value, schema, path) {
81	  var expectedType = schema.type;
82	
83	  if (expectedType === "array" && Array.isArray(value)) {
84	    return findInArray(value, schema, path);
85	  }
86	
87	  if (expectedType === "object" && value !== null && typeof value === "object" && !Array.isArray(value)) {
88	    return findInObject(value, schema, path);
89	  }
90	
91	  return [];
92	}
93	
94	function findInArray(value, schema, path) {
95	  var results = [];
96	  var itemSchema = schema.items || {};
97	
98	  // Track seen values for duplicate detection
99	  var seen = {};
100	
101	  for (var i = 0; i < value.length; i++) {
102	    var item = value[i];
103	    var itemPath = path + "[" + i + "]";
104	    var key = toComparable(item);
105	
106	    if (!seen[key]) {
107	      seen[key] = [];
108	    }
109	    seen[key].push(itemPath);
110	
111	    // Recurse into nested structures
112	    results = results.concat(findInValue(item, itemSchema, itemPath));
113	  }
114	
115	  // Report duplicates
116	  for (var k in seen) {
117	    if (seen[k].length > 1) {
118	      results.push(new DuplicateInfo(fromComparable(k), seen[k]));
119	    }
120	  }
121	
122	  return results;
123	}

125	function findInObject(value, schema, path) {
126	  var results = [];
127	  var props = schema.properties || {};
128	
129	  // Group values by their schema definition for cross-property dedup
130	  var schemaGroups = {};
131	
132	  var keys = Object.keys(value);
133	  for (var i = 0; i < keys.length; i++) {
134	    var fieldName = keys[i];
135	    var fieldValue = value[fieldName];
136	    var fieldPath = path + "." + fieldName;
137	    var fieldSchema = props[fieldName] || {};
138	
139	    // Recurse into nested structures
140	    results = results.concat(findInValue(fieldValue, fieldSchema, fieldPath));
141	
142	    // Group by schema key
143	    var schemaKey = toComparable(fieldSchema);
144	    if (!schemaGroups[schemaKey]) {
145	      schemaGroups[schemaKey] = {};
146	    }
147	    var comparable = toComparable(fieldValue);
148	    if (!schemaGroups[schemaKey][comparable]) {
149	      schemaGroups[schemaKey][comparable] = [];
150	    }
151	    schemaGroups[schemaKey][comparable].push(fieldPath);
152	  }
153	
154	  // Report duplicates across properties with the same schema
155	  for (var sg in schemaGroups) {
156	    var valueMap = schemaGroups[sg];
157	    for (var cv in valueMap) {
158	      if (valueMap[cv].length > 1) {
159	        results.push(new DuplicateInfo(fromComparable(cv), valueMap[cv]));
160	      }
161	    }
162	  }
163	
164	  return results;
165	}

167	/**
168	 * Find duplicates in a JSON or JSONL file against a schema.
169	 *
170	 * @param {string} dataPath - Path to the data file (JSON array or JSONL).
171	 * @param {string} schemaPath - Path to the schema JSON file.
172	 * @returns {DuplicateInfo[]}
173	 */
174	function findDuplicatesFromFile(dataPath, schemaPath) {
175	  var fs = require("fs");
176	  var schema = JSON.parse(fs.readFileSync(schemaPath, "utf-8"));
177	  var raw = fs.readFileSync(dataPath, "utf-8").trim();
178	
179	  if (!raw) return [];
180	
181	  var data;
182	  try {
183	    data = JSON.parse(raw);
184	  } catch (e) {
185	    // Try JSONL
186	    data = raw
187	      .split("\n")
188	      .filter(function (line) { return line.trim(); })
189	      .map(function (line) { return JSON.parse(line); });
190	  }
191	
192	  if (Array.isArray(data)) {
193	    return findInArray(data, { type: "array", items: schema }, "$");
194	  }
195	
196	  return findInValue(data, schema, "$");
197	}
