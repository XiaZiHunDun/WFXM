31	TOKEN_PATTERNS = [
32	    (r"\s+", None),              # skip whitespace
33	    (r"\d+(?:\.\d+)?", NUMBER),  # integer or decimal
34	    (r"\*\*", POW),              # must come before single *

47	class ParseError(Exception):
48	    """Raised when an expression is malformed."""

52	# Tokenizer

55	def _tokenize(expr: str) -> list[tuple[str, str]]:
56	    """Tokenize an expression string into (type, value) pairs."""
57	    tokens: list[tuple[str, str]] = []
58	    pos = 0
59	    while pos < len(expr):
60	        match = None
61	        for pattern, tok_type in _PATTERNS:
62	            m = pattern.match(expr, pos)
63	            if m:
64	                if tok_type is not None:
65	                    tokens.append((tok_type, m.group()))
66	                pos = m.end()
67	                match = True
68	                break
69	        if not match:
70	            raise ParseError(f"Unexpected character: {expr[pos]!r} at position {pos}")
71	    tokens.append((EOF, ""))
72	    return tokens

79	class _Parser:
80	    """Recursive descent parser for arithmetic expressions."""
81	
82	    def __init__(self, tokens: list[tuple[str, str]]):

84	        self.pos = 0

86	    def _current(self) -> tuple[str, str]:

89	    def _eat(self, tok_type: str) -> tuple[str, str]:
90	        cur = self._current()
91	        if cur[0] != tok_type:

93	                f"Expected {tok_type} but got {cur[0]} ({cur[1]!r}) at token {self.pos}"
94	            )

98	    # expression = term (('+' | '-') term)*

100	        result = self.term()
101	        while self._current()[0] in (PLUS, MINUS):

139	    # unary = ('+' | '-') unary | primary
140	    def unary(self) -> float:
141	        tok = self._current()

143	            self.pos += 1
144	            return -self.unary()
145	        if tok[0] == PLUS:
146	            self.pos += 1
147	            return self.unary()
148	        return self.primary()
149	
150	    # primary = NUMBER | '(' expression ')'
151	    def primary(self) -> float:
152	        tok = self._current()
153	        if tok[0] == NUMBER:
154	            self.pos += 1

156	        if tok[0] == LPAREN:
