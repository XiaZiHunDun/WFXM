34	def _int_to_roman(n):
35	    if not isinstance(n, int) or isinstance(n, bool):
36	        raise ValueError(f"Expected an integer, got {type(n).__name__}")
37	    if not 1 <= n <= 3999:
38	        raise ValueError(f"Integer must be between 1 and 3999, got {n}")
39	    result = []
40	    for value, numeral in ROMAN_VALUES:
41	        while n >= value:
42	            result.append(numeral)
43	            n -= value
44	    return "".join(result)

47	def _roman_to_int(s):
48	    if not isinstance(s, str) or not s:
49	        raise ValueError("Expected a non-empty string")
50	    s = s.upper().strip()
51	    for ch in s:
52	        if ch not in NUMERAL_MAP:
53	            raise ValueError(f"Invalid Roman numeral character: '{ch}'")
54	
55	    result = 0
56	    i = 0
57	    while i < len(s):
58	        # Check for subtractive pair (e.g., IV, IX, XL, etc.)
59	        if i + 1 < len(s) and NUMERAL_MAP[s[i]] < NUMERAL_MAP[s[i + 1]]:
60	            result += NUMERAL_MAP[s[i + 1]] - NUMERAL_MAP[s[i]]
61	            i += 2
62	        else:
63	            result += NUMERAL_MAP[s[i]]
64	            i += 1
65	    return result

6	def parse_expression(data: dict, parent_key: str = "", sep: str = ".") -> dict:
7	    """Flatten a nested dictionary into dot-notation keys.
8	
9	    Args:
10	        data: The nested dictionary to flatten.
11	        parent_key: Prefix for keys (used internally for recursion).
12	        sep: Separator between key segments.
13	
14	    Returns:
15	        A flat dictionary with dot-joined keys.
16	    """
17	    items: list[tuple[str, object]] = []
18	    for key, value in data.items():
19	        new_key = f"{parent_key}{sep}{key}" if parent_key != "" else str(key)
20	        if isinstance(value, Mapping):
21	            items.extend(parse_expression(value, new_key, sep=sep).items())
22	        else:
23	            items.append((new_key, value))
24	    return dict(items)
