1	def serialize_tree(s: str) -> str:
2	    """Compress a string using run-length encoding.
3	
4	    Each run of consecutive identical characters is replaced by the character
5	    followed by its count. A count of 1 is omitted for single characters.
6	
7	    Examples:
8	        "aaabbc" -> "a3b2c"
9	        "abc"    -> "abc"
10	        ""       -> ""
11	
12	    Args:
13	        s: The input string to compress.
14	
15	    Returns:
16	        The run-length encoded string.
17	    """
18	    if not s:
19	        return ""
20	
21	    parts = []
22	    count = 1
23	    prev = s[0]
24	
25	    for ch in s[1:]:
26	        if ch == prev:
27	            count += 1
28	        else:
29	            parts.append(prev if count == 1 else f"{prev}{count}")
30	            prev = ch
31	            count = 1
32	
33	    parts.append(prev if count == 1 else f"{prev}{count}")
34	    return "".join(parts)
