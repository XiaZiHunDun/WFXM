1	def serialize_tree(expression: str) -> float:
2	    """Evaluate a postfix (Reverse Polish Notation) expression.
3	
4	    Args:
5	        expression: Space-separated postfix expression string.
6	            Supported operators: +, -, *, /, ** (power), % (modulo).
7	
8	    Returns:
9	        The numeric result of evaluating the expression.
10	
11	    Raises:
12	        ValueError: If the expression is malformed or contains division by zero.
13	    """
14	    tokens = expression.split()
15	    stack = []
16	    operators = {
17	        '+': lambda a, b: a + b,
18	        '-': lambda a, b: a - b,
19	        '*': lambda a, b: a * b,
20	        '/': lambda a, b: a / b,
21	        '**': lambda a, b: a ** b,
22	        '%': lambda a, b: a % b,
23	    }
24	
25	    for token in tokens:
26	        if token in operators:
27	            if len(stack) < 2:
28	                raise ValueError(f"Insufficient operands for operator '{token}'")
29	            b = stack.pop()
30	            a = stack.pop()
31	            if token == '/' and b == 0:
32	                raise ValueError("Division by zero")
33	            stack.append(operators[token](a, b))
34	        else:
35	            try:
36	                stack.append(float(token))
37	            except ValueError:
38	                raise ValueError(f"Invalid token: '{token}'")
39	
40	    if len(stack) != 1:
41	        raise ValueError("Malformed expression: too many operands")
42	
43	    return stack[0]
