1	/**
2	 * Converts an infix expression to postfix (Reverse Polish Notation)
3	 * using the Shunting-yard algorithm.
4	 *
5	 * Supports: +, -, *, /, ^, parentheses, multi-digit numbers, and decimals.
6	 */
7	function infixToPostfix(expression) {
8	  const precedence = { '+': 1, '-': 1, '*': 2, '/': 2, '^': 3 };
9	  const rightAssociative = new Set(['^']);
10	
11	  const output = [];
12	  const operatorStack = [];
13	
14	  // Tokenize: numbers (including decimals) and operators/parens
15	  const tokens = expression.match(/\d+(\.\d+)?|[+\-*/^()]/g);
16	  if (!tokens) return '';
17	
18	  for (const token of tokens) {
19	    if (/^\d/.test(token)) {
20	      output.push(token);
21	    } else if (token === '(') {
22	      operatorStack.push(token);
23	    } else if (token === ')') {
24	      while (operatorStack.length && operatorStack[operatorStack.length - 1] !== '(') {
25	        output.push(operatorStack.pop());
26	      }
27	      operatorStack.pop(); // discard '('
28	    } else {
29	      // operator
30	      while (
31	        operatorStack.length &&
32	        operatorStack[operatorStack.length - 1] !== '(' &&
33	        (precedence[operatorStack[operatorStack.length - 1]] > precedence[token] ||
34	          (precedence[operatorStack[operatorStack.length - 1]] === precedence[token] &&
35	            !rightAssociative.has(token)))
36	      ) {
37	        output.push(operatorStack.pop());
38	      }
39	      operatorStack.push(token);
40	    }
41	  }
42	
43	  while (operatorStack.length) {
44	    output.push(operatorStack.pop());
45	  }
46	
47	  return output.join(' ');
48	}
