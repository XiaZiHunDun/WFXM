1	"""Depth-First Search (DFS) implementation for a graph.
2	
3	DFS explores a graph by going as deep as possible along each branch
4	before backtracking.  Think of it like navigating a maze: you walk
5	forward until you hit a dead end, then retrace your steps to the
6	last junction and try the next unexplored path.
7	
8	Two implementations are provided:
9	  1. Iterative — uses an explicit stack (Python list used as LIFO).
10	  2. Recursive — uses the call stack implicitly.
11	
12	Both visit vertices in the same order for the same adjacency-list ordering.
13	
14	A brute-force path enumeration is also included for comparison.
15	"""
16	
17	from collections import defaultdict
18	
19	
20	class Graph:
21	    """Graph stored as an adjacency list.
22	
23	    Each vertex maps to a list of its neighbors.
24	    An optional `directed` flag controls whether edges are one-way or two-way.
25	    """
26	
27	    def __init__(self, directed=False):
28	        # defaultdict(list) means a missing key auto-creates an empty list,
29	        # so we never get a KeyError when accessing a new vertex.
30	        self.adj = defaultdict(list)
31	        self.directed = directed
32	
33	    def add_edge(self, u, v):
34	        """Add an edge from u to v.  For undirected graphs, also add v to u."""
35	        self.adj[u].append(v)
36	        if not self.directed:
37	            # Undirected: the edge goes both ways
38	            self.adj[v].append(u)
39	
40	    def dfs(self, start, target=None):
41	        """Iterative DFS using an explicit stack.
42	
43	        Args:
44	            start:  The vertex to begin the search from.
45	            target: Optional vertex to search for.  If provided the method
46	                    returns True/False; otherwise it returns the full set
47	                    of visited vertices.
48	
49	        How it works:
50	            1. Push `start` onto the stack.
51	            2. Pop the top vertex — this is the LIFO (last-in, first-out)
52	               behaviour that gives DFS its "depth-first" character.
53	            3. If we've already visited it, skip (a vertex can be pushed
54	               multiple times by different neighbors before being processed).
55	            4. Mark it visited.  If it's the target, we're done.
56	            5. Push all unvisited neighbors onto the stack.  We reverse the
57	               neighbor list so that the first neighbor in the adjacency list
58	               is processed first — this matches the order of recursive DFS.
59	            6. Repeat until the stack is empty (all reachable vertices visited).
60	        """
61	        # Edge case: start vertex has no outgoing edges and isn't the target
62	        if start not in self.adj and target is not None and start != target:
63	            return False
64	
65	        visited = set()     # Tracks vertices we've already processed
66	        stack = [start]     # The stack — initialized with the start vertex
67	
68	        while stack:
69	            # Step 2: Pop the most recently added vertex (LIFO)
70	            node = stack.pop()
71	
72	            # Step 3: Skip if already visited (duplicate on the stack)
73	            if node in visited:
74	                continue
75	
76	            # Step 4: Mark as visited
77	            visited.add(node)
78	
79	            # Step 5: Early exit if we found the target
80	            if target is not None and node == target:
81	                return True
82	
83	            # Step 6: Push unvisited neighbors.
84	            # `reversed` ensures we process the first-listed neighbor first,
85	            # matching the natural left-to-right order of recursive DFS.
86	            for neighbor in reversed(self.adj[node]):
87	                if neighbor not in visited:
88	                    stack.append(neighbor)
89	
90	        # Stack exhausted — either return the visited set or report "not found"
91	        return False if target is not None else visited
92	
93	    def dfs_recursive(self, start, target=None):
94	        """Recursive DFS using the call stack.
95	
96	        This is the textbook version: visit a node, then recurse into each
97	        unvisited neighbor.  The call stack acts as the implicit stack.
98	
99	        Args:
100	            start:  The vertex to begin from.
101	            target: Optional target vertex.
102	
103	        Returns:
104	            The set of visited vertices, or True/False when searching for a target.
105	        """
106	        visited = set()
107	
108	        def _dfs(node):
109	            # Mark the current node as visited before exploring its neighbors
110	            visited.add(node)
111	
112	            # If we're looking for a specific vertex, check now
113	            if target is not None and node == target:
114	                return True
115	
116	            # Recurse into each unvisited neighbor.
117	            # Because we recurse immediately, we go *deep* before going *wide*.
118	            for neighbor in self.adj[node]:
119	                if neighbor not in visited:
120	                    if _dfs(neighbor):
121	                        return True  # Propagate "found" up the call stack
122	
123	            return False
124	
125	        result = _dfs(start)
126	        # When no target, return the full visited set; otherwise the boolean
127	        return result if target is not None else visited
128	
129	    def dfs_all(self):
130	        """Run DFS across every connected component in the graph.
131	
132	        This handles disconnected graphs: we iterate over every vertex and
133	        start a new DFS whenever we find one that hasn't been visited yet.
134	        Each DFS run discovers exactly one connected component.
135	
136	        Returns:
137	            A list of sets, where each set is one connected component.
138	        """
139	        visited = set()
140	        components = []
141	
142	        # Iterate over a snapshot of keys (`list(self.adj)`) because we
143	        # don't modify the dict during iteration — just reading.
144	        for node in list(self.adj):
145	            if node in visited:
146	                continue  # Already part of a previous component
147	
148	            # Start a new DFS from this unvisited node
149	            component = set()
150	            stack = [node]
151	
152	            while stack:
153	                curr = stack.pop()
154	                if curr in visited:
155	                    continue
156	                visited.add(curr)
157	                component.add(curr)
158	
159	                # Push neighbors to continue expanding this component
160	                for neighbor in self.adj[curr]:
161	                    if neighbor not in visited:
162	                        stack.append(neighbor)
163	
164	            # This component is fully explored
165	            components.append(component)
166	
167	        return components
168

170	def brute_force_path_exists(graph, start, target, max_depth=20):
171	    """Brute-force: try all walks from start to target without a visited set.
172	
173	    Unlike DFS, this does NOT track which vertices have been visited.
174	    It explores every possible walk (including revisiting nodes), up to a
175	    depth limit.  This means it explores an exponential number of states.
176	
177	    Time complexity: O(B^d) where B = max branching factor, d = depth limit.
178	    For a graph with average degree k, this is O(k^max_depth) — exponential.
179	
180	    Contrast with DFS: DFS marks vertices visited and never re-explores them,
181	    giving O(V + E).  Brute-force revisits the same vertices many times.
182	    """
183	    def _walk(node, depth):
184	        if depth > max_depth:
185	            return False
186	        if node == target:
187	            return True
188	        for neighbor in graph.adj[node]:
189	            if _walk(neighbor, depth + 1):
190	                return True
191	        return False
192	
193	    return _walk(start, 0)
194	
195	
196	def brute_force_all_paths(graph, start, target):
197	    """Brute-force: return every simple path from start to target.
198	
199	    Returns a list of paths (each path is a list of vertices).
200	    This is the fully exhaustive version — it doesn't stop at the first match.
201	    """
202	    results = []
203	
204	    def _paths(node, path, visited):
205	        if node == target:
206	            results.append(list(path))
207	            return
208	        for neighbor in graph.adj[node]:
209	            if neighbor not in visited:
210	                visited.add(neighbor)
211	                path.append(neighbor)
212	                _paths(neighbor, path, visited)
213	                path.pop()
214	                visited.remove(neighbor)
215	
216	    _paths(start, [start], {start})
217	    return results
