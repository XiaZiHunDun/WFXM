1	"""
2	Bellman-Ford Shortest Path Algorithm
3	=====================================
4	
5	Finds the shortest paths from a single source vertex to all other vertices
6	in a weighted directed graph. Unlike Dijkstra's algorithm, Bellman-Ford
7	handles negative edge weights and can detect negative-weight cycles.
8	
9	Time complexity:  O(V * E)
10	Space complexity: O(V)
11	"""
12	
13	import itertools
14	import time
15	import random

18	def bellman_ford(vertices, edges, source):
19	    """
20	    Compute shortest paths from `source` to every other vertex.
21	
22	    Args:
23	        vertices: number of vertices, labeled 0 .. vertices-1
24	        edges:    list of (u, v, weight) directed edges
25	        source:   starting vertex
26	
27	    Returns:
28	        dist:             list where dist[v] is the shortest distance from source to v
29	        pred:             list where pred[v] is the predecessor of v on the shortest path
30	        has_negative_cycle: True if a negative-weight cycle is reachable from source
31	    """
32	
33	    # ---------------------------------------------------------------
34	    # STEP 1 — Initialization
35	    #
36	    # Set the distance to every vertex to infinity (we haven't found
37	    # any paths yet).  The source's distance is 0 because we're
38	    # already there.  The predecessor of every vertex is "unknown"
39	    # (represented by -1).
40	    # ---------------------------------------------------------------
41	    dist = [float("inf")] * vertices
42	    pred = [-1] * vertices
43	    dist[source] = 0
44	
45	    # ---------------------------------------------------------------
46	    # STEP 2 — Edge relaxation (repeated V-1 times)
47	    #
48	    # Why V-1?  A shortest path that visits every vertex at most once
49	    # uses at most V-1 edges.  Each pass over all edges can extend
50	    # the shortest known path by one edge, so after V-1 passes every
51	    # shortest path has been fully discovered.
52	    #
53	    # "Relaxing" edge (u, v, w) means: if we already know a path to
54	    # u, check whether going u -> v yields a shorter path to v.
55	    # If dist[u] + w < dist[v], we found a better route — update it.
56	    #
57	    # Optimization: if a full pass makes zero updates, the algorithm
58	    # has converged early and we can stop.
59	    # ---------------------------------------------------------------
60	    for i in range(vertices - 1):
61	        updated = False
62	        for u, v, w in edges:
63	            # Only relax if u is reachable from the source.
64	            if dist[u] != float("inf") and dist[u] + w < dist[v]:
65	                dist[v] = dist[u] + w
66	                pred[v] = u
67	                updated = True
68	        if not updated:
69	            break  # no improvements this pass — we're done
70	
71	    # ---------------------------------------------------------------
72	    # STEP 3 — Negative-weight cycle detection
73	    #
74	    # After V-1 passes the distances are final *unless* a
75	    # negative-weight cycle exists.  If we can still relax an edge
76	    # on one more pass, a cycle must be shrinking the path cost
77	    # indefinitely, meaning no true shortest path exists.
78	    # ---------------------------------------------------------------
79	    has_negative_cycle = False
80	    for u, v, w in edges:
81	        if dist[u] != float("inf") and dist[u] + w < dist[v]:
82	            has_negative_cycle = True
83	            break
84	
85	    return dist, pred, has_negative_cycle

88	def reconstruct_path(pred, source, target):
89	    """
90	    Walk the predecessor chain backwards from target to source
91	    to recover the actual shortest path as a list of vertices.
92	    Returns an empty list if target is unreachable.
93	    """
94	    path = []
95	    current = target
96	    while current != -1:
97	        path.append(current)
98	        if current == source:
99	            return path[::-1]  # reverse to get source -> target order
100	        current = pred[current]
101	    return []  # reached a vertex with no predecessor — unreachable
102

104	# ---------------------------------------------------------------------------
105	# Brute-force shortest path — O(V!)  (for verification on small graphs)
106	# ---------------------------------------------------------------------------
107	def brute_force_shortest_path(vertices, edges, source):
108	    """
109	    Enumerate every simple path from source to every other vertex.
110	    Only works for tiny graphs (factorial complexity).
111	    Returns a dist list in the same format as bellman_ford.
112	    """
113	    # Build adjacency list for fast neighbor lookup
114	    adj = [[] for _ in range(vertices)]
115	    for u, v, w in edges:
116	        adj[u].append((v, w))
117	
118	    dist = [float("inf")] * vertices
119	    dist[source] = 0
120	
121	    for target in range(vertices):
122	        if target == source:
123	            continue
124	        # Try every permutation of intermediate vertices (0 to all of them)
125	        intermediates = [v for v in range(vertices) if v != source and v != target]
126	        best = float("inf")
127	        for k in range(len(intermediates) + 1):
128	            for subset in itertools.permutations(intermediates, k):
129	                path = [source] + list(subset) + [target]
130	                # Walk the path, summing edge weights; bail if an edge is missing
131	                cost = 0
132	                valid = True
133	                for i in range(len(path) - 1):
134	                    u, v = path[i], path[i + 1]
135	                    edge_weight = None
136	                    for neighbor, w in adj[u]:
137	                        if neighbor == v:
138	                            edge_weight = w
139	                            break
140	                    if edge_weight is None:
141	                        valid = False
142	                        break
143	                    cost += edge_weight
144	                if valid:
145	                    best = min(best, cost)
146	        dist[target] = best
147	
148	    return dist

151	def random_graph(num_vertices, num_edges, weight_range=(1, 20)):
152	    """Generate a random directed graph with the given number of edges."""
153	    edges = set()
154	    while len(edges) < num_edges:
155	        u = random.randint(0, num_vertices - 1)
156	        v = random.randint(0, num_vertices - 1)
157	        if u != v:
158	            w = random.randint(*weight_range)
159	            edges.add((u, v, w))
160	    return list(edges)
