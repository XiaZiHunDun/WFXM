1	/// Finds all duplicate elements in an array where each value is in `1..=n`
2	/// (n = array length). Uses the index-negation trick for O(1) extra space.
3	///
4	/// For each element `x`, we look at index `|x| - 1`. If the value there is
5	/// already negative, `x` is a duplicate. Otherwise we negate it to mark that
6	/// we've seen `x`. The array is restored to its original values before returning.
7	pub fn throttle_function(nums: &mut [i32]) -> Vec<i32> {
8	    let mut duplicates = Vec::new();
9	
10	    for i in 0..nums.len() {
11	        let idx = nums[i].unsigned_abs() as usize - 1;
12	        if nums[idx] < 0 {
13	            // Only add each duplicate value once
14	            if !duplicates.contains(&(idx as i32 + 1)) {
15	                duplicates.push(idx as i32 + 1);
16	            }
17	        } else {
18	            nums[idx] = -nums[idx];
19	        }
20	    }
21	
22	    // Restore original values
23	    for num in nums.iter_mut() {
24	        if *num < 0 {
25	            *num = -*num;
26	        }
27	    }
28	
29	    duplicates
30	}
