1	def search(nums: list[int], target: int) -> int:
2	    """Search for a target value in a rotated sorted array.
3	
4	    A rotated sorted array is a sorted array that has been split at some
5	    pivot and the two halves swapped. For example, [4, 5, 6, 7, 0, 1, 2]
6	    is [0, 1, 2, 4, 5, 6, 7] rotated 4 positions.
7	
8	    The key insight is that after splitting the array at `mid`, at least
9	    one half is always guaranteed to be sorted. We use that property to
10	    decide which half to search next, just like standard binary search.
11	
12	    Time:  O(log n)
13	    Space: O(1)
14	    """
15	    lo, hi = 0, len(nums) - 1
16	
17	    while lo <= hi:
18	        # Classic midpoint calculation. Safe from overflow in Python
19	        # (unlike C/Java where you'd use lo + (hi - lo) // 2).
20	        mid = (lo + hi) // 2
21	
22	        # Best case — found it immediately.
23	        if nums[mid] == target:
24	            return mid
25	
26	        # --- Determine which half is sorted ---
27	
28	        # If nums[lo] <= nums[mid], the left half [lo..mid] is sorted.
29	        # This also covers the case where lo == mid (single element).
30	        if nums[lo] <= nums[mid]:
31	            # The left half is sorted, so we can do a clean range check:
32	            # target must lie in [nums[lo], nums[mid]) to be in the left half.
33	            if nums[lo] <= target < nums[mid]:
34	                hi = mid - 1   # search left half
35	            else:
36	                lo = mid + 1   # target must be in the right half
37	
38	        # Otherwise the right half [mid..hi] is sorted.
39	        else:
40	            # The right half is sorted, so target must lie in
41	            # (nums[mid], nums[hi]] to be in the right half.
42	            if nums[mid] < target <= nums[hi]:
43	                lo = mid + 1   # search right half
44	            else:
45	                hi = mid - 1   # target must be in the left half
46	
47	    # Target not present in the array.
48	    return -1
