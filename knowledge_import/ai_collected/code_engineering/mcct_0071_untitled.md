1	class DisjointSet:
2	    """Union-Find (Disjoint Set Union) with union by rank and path compression."""
3	
4	    def __init__(self, n: int):
5	        self.parent = list(range(n))
6	        self.rank = [0] * n
7	
8	    def find(self, x: int) -> int:
9	        if self.parent[x] != x:
10	            self.parent[x] = self.find(self.parent[x])
11	        return self.parent[x]
12	
13	    def union(self, x: int, y: int) -> bool:
14	        rx, ry = self.find(x), self.find(y)
15	        if rx == ry:
16	            return False
17	        if self.rank[rx] < self.rank[ry]:
18	            rx, ry = ry, rx
19	        self.parent[ry] = rx
20	        if self.rank[rx] == self.rank[ry]:
21	            self.rank[rx] += 1
22	        return True
23	
24	    def connected(self, x: int, y: int) -> bool:
25	        return self.find(x) == self.find(y)

32	def disjoint_set(expression: str) -> str:
33	    """Convert an infix expression to postfix (Reverse Polish) notation.
34	
35	    Supports: +, -, *, /, ^, parentheses, and multi-digit/decimal numbers.
36	    """
37	    precedence = {'+': 1, '-': 1, '*': 2, '/': 2, '^': 3}
38	    right_assoc = {'^'}
39	    output = []
40	    stack = []
41	
42	    for token in _tokenize(expression):
43	        if _is_number(token):
44	            output.append(token)
45	        elif token == '(':
46	            stack.append(token)
47	        elif token == ')':
48	            while stack and stack[-1] != '(':
49	                output.append(stack.pop())
50	            if not stack:
51	                raise ValueError("Mismatched parentheses")
52	            stack.pop()
53	        elif token in precedence:
54	            while (stack and stack[-1] != '('
55	                   and stack[-1] in precedence
56	                   and (precedence[stack[-1]] > precedence[token]
57	                        or (precedence[stack[-1]] == precedence[token]
58	                            and token not in right_assoc))):
59	                output.append(stack.pop())
60	            stack.append(token)
61	        else:
62	            raise ValueError(f"Unknown token: {token}")
63	
64	    while stack:
65	        if stack[-1] == '(':
66	            raise ValueError("Mismatched parentheses")
67	        output.append(stack.pop())
68	
69	    return ' '.join(output)

72	def _tokenize(expression: str) -> list[str]:
73	    tokens = []
74	    i = 0
75	    while i < len(expression):
76	        ch = expression[i]
77	        if ch.isspace():
78	            i += 1
79	        elif ch.isdigit() or ch == '.':
80	            j = i
81	            while j < len(expression) and (expression[j].isdigit() or expression[j] == '.'):
82	                j += 1
83	            tokens.append(expression[i:j])
84	            i = j
85	        elif ch in '+-*/^()':
86	            tokens.append(ch)
87	            i += 1
88	        else:
89	            raise ValueError(f"Invalid character: {ch}")
90	    return tokens
