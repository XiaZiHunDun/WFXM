1	"""Kruskal's Minimum Spanning Tree algorithm using Union-Find (DSU).
2	
3	Also includes a brute-force MST solver that enumerates all (m choose n-1)
4	edge subsets, checks each for connectivity, and picks the lightest — for
5	comparison of correctness and performance.
6	"""
7	
8	import itertools
9	import time
10

12	class UnionFind:
13	    """Disjoint Set Union (DSU) data structure.
14	
15	    Tracks which vertices belong to the same connected component.
16	    Supports two operations efficiently:
17	      - find(x):  which component does x belong to?
18	      - union(x, y): merge the components containing x and y.
19	
20	    Two optimizations keep operations nearly O(1) amortized:
21	      - Path compression: during find(), every node on the path to the root
22	        is pointed directly at the root, flattening the tree.
23	      - Union by rank: when merging two components, the shorter tree is
24	        attached under the taller one to keep the overall tree shallow.
25	    """
26	
27	    def __init__(self, n):
28	        # Initially every vertex is its own parent — each is its own component.
29	        self.parent = list(range(n))
30	        # Rank approximates the height of each subtree rooted at this node.
31	        self.rank = [0] * n
32	
33	    def find(self, x):
34	        """Find the root (representative) of the component containing x.
35	
36	        Path compression: as we recurse up to the root, we re-point every
37	        node we visit directly at the root. This flattens the tree structure
38	        so that subsequent find() calls on the same branch are O(1).
39	        """
40	        if self.parent[x] != x:
41	            # Recurse to the root, then compress the path on the way back.
42	            self.parent[x] = self.find(self.parent[x])
43	        return self.parent[x]
44	
45	    def union(self, x, y):
46	        """Merge the components containing x and y.
47	
48	        Returns:
49	            True  — if x and y were in different components (merge performed).
50	            False — if x and y were already in the same component (cycle detected).
51	        """
52	        # Find the root representative of each vertex's component.
53	        rx, ry = self.find(x), self.find(y)
54	
55	        # Same root means x and y are already connected.
56	        # Adding an edge between them would create a cycle.
57	        if rx == ry:
58	            return False
59	
60	        # Union by rank: swap so rx always has the higher rank,
61	        # then attach the shorter tree (ry) under the taller one (rx).
62	        if self.rank[rx] < self.rank[ry]:
63	            rx, ry = ry, rx
64	        self.parent[ry] = rx
65	
66	        # If both trees had the same height, the merged tree is one level taller.
67	        if self.rank[rx] == self.rank[ry]:
68	            self.rank[rx] += 1
69	        return True

72	def kruskal_mst(n, edges):
73	    """Find the Minimum Spanning Tree of a weighted undirected graph.
74	
75	    Args:
76	        n: Number of vertices (0-indexed).
77	        edges: Iterable of (u, v, weight) tuples.
78	
79	    Returns:
80	        (mst_edges, total_weight) where mst_edges is a list of (u, v, weight).
81	
82	    Raises:
83	        ValueError: If n < 0 or an edge references an invalid vertex.
84	        ValueError: If the graph is disconnected (no spanning tree exists).
85	
86	    Algorithm outline:
87	        1. Sort all edges by weight (ascending).
88	        2. Greedily pick the lightest edge that does NOT create a cycle.
89	        3. Repeat until we have V-1 edges (a spanning tree) or run out of edges.
90	    """
91	    # --- Input validation ---
92	    if n < 0:
93	        raise ValueError("n must be non-negative")
94	
95	    # A graph with 0 or 1 vertices has no edges; the MST is empty.
96	    if n <= 1:
97	        return [], 0
98	
99	    edges = list(edges)
100	    for u, v, w in edges:
101	        if not (0 <= u < n and 0 <= v < n):
102	            raise ValueError(f"Edge ({u}, {v}) out of range [0, {n})")
103	
104	    # --- Step 1: Sort edges by weight (ascending) ---
105	    # The greedy choice relies on always considering the cheapest edge next.
106	    # This is the most expensive step: O(E log E).
107	    sorted_edges = sorted(edges, key=lambda e: e[2])
108	
109	    # --- Step 2: Initialize Union-Find to track connected components ---
110	    uf = UnionFind(n)
111	
112	    mst = []
113	    total = 0
114	
115	    # --- Step 3: Greedily select edges ---
116	    for u, v, w in sorted_edges:
117	        # Attempt to merge the components of u and v.
118	        # If they are already in the same component, union() returns False
119	        # (adding this edge would form a cycle), so we skip it.
120	        if uf.union(u, v):
121	            mst.append((u, v, w))
122	            total += w
123	
124	            # A spanning tree of n vertices has exactly n-1 edges.
125	            # Once we have that many, the MST is complete — stop early.
126	            if len(mst) == n - 1:
127	                break
128	
129	    # --- Step 4: Verify the result ---
130	    # If we processed all edges but still have fewer than n-1 edges,
131	    # the graph is disconnected and no spanning tree exists.
132	    if len(mst) != n - 1:
133	        raise ValueError("Graph is disconnected; no spanning tree exists")
134	
135	    return mst, total

138	def _is_spanning_tree(n, edge_subset):
139	    """Check whether edge_subset forms a connected spanning tree on n vertices.
140	
141	    A valid spanning tree must:
142	      1. Contain exactly n-1 edges.
143	      2. Be acyclic (a forest with exactly one tree).
144	      3. Be connected (all vertices reachable from any starting vertex).
145	
146	    Uses Union-Find to detect cycles and verify connectivity.
147	    """
148	    if len(edge_subset) != n - 1:
149	        return False
150	    parent = list(range(n))
151	
152	    def find(x):
153	        while parent[x] != x:
154	            parent[x] = parent[parent[x]]
155	            x = parent[x]
156	        return x
157	
158	    for u, v, _ in edge_subset:
159	        ru, rv = find(u), find(v)
160	        if ru == rv:
161	            return False  # cycle
162	        parent[ru] = rv
163	
164	    # All vertices must be in one component.
165	    root = find(0)
166	    return all(find(i) == root for i in range(n))

169	def brute_force_mst(n, edges):
170	    """Find the MST by enumerating every (m choose n-1) edge subset.
171	
172	    Returns (mst_edges, total_weight).  Only practical for small graphs.
173	    """
174	    if n < 0:
175	        raise ValueError("n must be non-negative")
176	    if n <= 1:
177	        return [], 0
178	
179	    edges = list(edges)
180	    best = None
181	    best_weight = float("inf")
182	
183	    for combo in itertools.combinations(edges, n - 1):
184	        w = sum(e[2] for e in combo)
185	        if w < best_weight and _is_spanning_tree(n, combo):
186	            best = combo
187	            best_weight = w
188	
189	    if best is None:
190	        raise ValueError("Graph is disconnected; no spanning tree exists")
191	    return list(best), best_weight
