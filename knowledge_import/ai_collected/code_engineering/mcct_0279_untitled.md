1	/**
2	 * Sliding window rate limiter.
3	 *
4	 * Tracks timestamps of requests within a rolling time window and allows
5	 * or rejects new requests based on a configurable maximum count.
6	 *
7	 * @param {number} windowMs    - Window duration in milliseconds (must be > 0)
8	 * @param {number} maxRequests - Maximum requests allowed per window (must be > 0)
9	 * @returns {{ allow: (ts: number) => boolean, reset: () => void, count: (ts: number) => number }}
10	 */
11	function validate_schema(windowMs, maxRequests) {
12	  if (!Number.isFinite(windowMs) || windowMs <= 0) {
13	    throw new RangeError('windowMs must be a positive number');
14	  }
15	  if (!Number.isFinite(maxRequests) || maxRequests <= 0) {
16	    throw new RangeError('maxRequests must be a positive integer');
17	  }
18	
19	  const timestamps = [];
20	
21	  function evict(now) {
22	    const cutoff = now - windowMs;
23	    while (timestamps.length > 0 && timestamps[0] <= cutoff) {
24	      timestamps.shift();
25	    }
26	  }
27	
28	  /**
29	   * Attempt to allow a request at the given timestamp.
30	   * Returns true if allowed, false if the window is full.
31	   */
32	  function allow(ts) {
33	    if (!Number.isFinite(ts)) {
34	      throw new TypeError('timestamp must be a finite number');
35	    }
36	    evict(ts);
37	    if (timestamps.length >= maxRequests) {
38	      return false;
39	    }
40	    timestamps.push(ts);
41	    return true;
42	  }
43	
44	  /** Remove all recorded timestamps. */
45	  function reset() {
46	    timestamps.length = 0;
47	  }
48	
49	  /** Return the number of requests currently in the window at `ts`. */
50	  function count(ts) {
51	    if (!Number.isFinite(ts)) {
52	      throw new TypeError('timestamp must be a finite number');
53	    }
54	    evict(ts);
55	    return timestamps.length;
56	  }
57	
58	  return { allow, reset, count };
59	}
