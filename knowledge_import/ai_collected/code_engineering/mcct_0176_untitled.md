23	    """Raised when a resource is not found (404)."""
24	
25	
26	class RateLimitError(AirtableError):
27	    """Raised when rate limited (429)."""
28	
29	    def __init__(self, message: str, retry_after: float | None = None) -> None:
30	        super().__init__(message, status_code=429)
31	        self.retry_after = retry_after
32	
33	
34	class InvalidRequestError(AirtableError):
35	    """Raised for 422 Unprocessable Entity."""
36	
37

41	
42	_STATUS_MAP: dict[int, type[AirtableError]] = {
43	    401: AuthenticationError,
44	    403: ForbiddenError,
45	    404: NotFoundError,
46	    422: InvalidRequestError,
47	}
48	
49	
50	def raise_for_status(status_code: int, message: str, retry_after: float | None = None) -> None:
51	    """Raise the appropriate exception for a given HTTP status code."""
52	    if status_code == 429:
53	        raise RateLimitError(message, retry_after=retry_after)
54	    exc_cls = _STATUS_MAP.get(status_code)
55	    if exc_cls is not None:
56	        raise exc_cls(message, status_code=status_code)
57	    if 500 <= status_code < 600:
58	        raise ServerError(message, status_code=status_code)
59	    raise AirtableError(message, status_code=status_code)
