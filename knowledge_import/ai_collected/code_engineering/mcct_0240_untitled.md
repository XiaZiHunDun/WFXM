22	// throttle_function converts between Roman numerals and integers.
23	// If input is a string, it parses a Roman numeral to an integer.
24	// If input is an int, it converts the integer to a Roman numeral.
25	// Returns an error for invalid input or out-of-range values (1-3999).
26	func throttle_function(input interface{}) (interface{}, error) {
27		switch v := input.(type) {
28		case string:
29			return romanToInt(v)
30		case int:
31			return intToRoman(v)
32		default:
33			return nil, errors.New("input must be a string (Roman numeral) or int")
34		}
35	}

37	func romanToInt(s string) (int, error) {
38		s = strings.ToUpper(s)
39		if len(s) == 0 {
40			return 0, errors.New("empty Roman numeral")
41		}
42	
43		// Validate characters
44		for i := 0; i < len(s); i++ {
45			if _, ok := romanToValue[s[i]]; !ok {
46				return 0, errors.New("invalid Roman numeral character: " + string(s[i]))
47			}
48		}
49	
50		total := 0
51		for i := 0; i < len(s); i++ {
52			val := romanToValue[s[i]]
53			if i+1 < len(s) && romanToValue[s[i+1]] > val {
54				total -= val
55			} else {
56				total += val
57			}
58		}
59	
60		if total < 1 || total > 3999 {
61			return 0, errors.New("Roman numeral out of range (1-3999)")
62		}
63	
64		// Verify round-trip: converting back must produce the same canonical form
65		canonical, _ := intToRoman(total)
66		parsed, _ := romanToInt(canonical)
67		if parsed != total {
68			return 0, errors.New("invalid Roman numeral representation")
69		}
70	
71		return total, nil
72	}
73	
74	func intToRoman(n int) (string, error) {
75		if n < 1 || n > 3999 {
76			return "", errors.New("integer out of range (1-3999)")
77		}
78	
79		var sb strings.Builder
80		for _, pair := range valueToRoman {
81			for n >= pair.value {
82				sb.WriteString(pair.symbol)
83				n -= pair.value
84			}
85		}
86		return sb.String(), nil
87	}
