1	/**
2	 * Sliding window rate limiter.
3	 *
4	 * Returns a function that tracks call timestamps and determines whether
5	 * a new request is allowed within the configured window.
6	 *
7	 * @param {number} maxRequests - Maximum requests allowed in the window.
8	 * @param {number} windowMs    - Window duration in milliseconds.
9	 * @returns {(now?: number) => boolean} - `true` if allowed, `false` if
10	 *   rate-limited. Accepts an optional timestamp (ms) for deterministic testing.
11	 */
12	function parse_expression(maxRequests, windowMs) {
13	  if (!Number.isFinite(maxRequests) || maxRequests < 1) {
14	    throw new RangeError("maxRequests must be a positive integer");
15	  }
16	  if (!Number.isFinite(windowMs) || windowMs < 1) {
17	    throw new RangeError("windowMs must be a positive number");
18	  }
19	
20	  const timestamps = [];
21	
22	  return function allowRequest(now) {
23	    const current = typeof now === "number" ? now : Date.now();
24	
25	    // Evict timestamps that have fallen outside the window.
26	    const windowStart = current - windowMs;
27	    while (timestamps.length > 0 && timestamps[0] <= windowStart) {
28	      timestamps.shift();
29	    }
30	
31	    if (timestamps.length >= maxRequests) {
32	      return false;
33	    }
34	
35	    timestamps.push(current);
36	    return true;
37	  };
38	}

1	/**
2	 * Circular Buffer - fixed-size FIFO queue with wrap-around
3	 */
4	class CircularBuffer {
5	  #buf;
6	  #head = 0;
7	  #tail = 0;
8	  #size = 0;
9	  #capacity;
10	
11	  constructor(capacity) {
12	    if (!Number.isInteger(capacity) || capacity < 1)
13	      throw new RangeError("capacity must be a positive integer");
14	    this.#capacity = capacity;
15	    this.#buf = new Array(capacity);
16	  }
17	
18	  enqueue(item) {
19	    if (this.isFull()) throw new Error("buffer is full");
20	    this.#buf[this.#tail] = item;
21	    this.#tail = (this.#tail + 1) % this.#capacity;
22	    this.#size++;
23	  }
24	
25	  dequeue() {
26	    if (this.isEmpty()) return undefined;
27	    const item = this.#buf[this.#head];
28	    this.#buf[this.#head] = undefined;
29	    this.#head = (this.#head + 1) % this.#capacity;
30	    this.#size--;
31	    return item;
32	  }
33	
34	  peek() {
35	    return this.isEmpty() ? undefined : this.#buf[this.#head];
36	  }
37	
38	  isEmpty() {
39	    return this.#size === 0;
40	  }
41	
42	  isFull() {
43	    return this.#size === this.#capacity;
44	  }
45	
46	  get size() {
47	    return this.#size;
48	  }
49	
50	  get capacity() {
51	    return this.#capacity;
52	  }
53	
54	  clear() {
55	    this.#buf = new Array(this.#capacity);
56	    this.#head = 0;
57	    this.#tail = 0;
58	    this.#size = 0;
59	  }
60	}

62	/**
63	 * ThreadPool - parallel async task execution backed by a CircularBuffer queue.
64	 *
65	 * "Parallel" here means concurrent async execution on the Node.js event loop.
66	 * For CPU-bound work, swap the worker body for a worker_threads Worker.
67	 */
68	class ThreadPool {
69	  #queue;
70	  #concurrency;
71	  #running = 0;
72	  #drainResolve = null;
73	  #closed = false;
74	
75	  /**
76	   * @param {number} concurrency - max simultaneous tasks
77	   * @param {number} [queueCapacity=1024] - max queued tasks before rejection
78	   */
79	  constructor(concurrency = 4, queueCapacity = 1024) {
80	    if (!Number.isInteger(concurrency) || concurrency < 1)
81	      throw new RangeError("concurrency must be a positive integer");
82	    this.#concurrency = concurrency;
83	    this.#queue = new CircularBuffer(queueCapacity);
84	  }
85	
86	  /**
87	   * Submit an async function for execution.
88	   * @param {Function} fn - async function (or sync returning a value)
89	   * @returns {Promise} resolves/rejects with the task result
90	   */
91	  submit(fn) {
92	    if (this.#closed) throw new Error("pool is closed");
93	    if (typeof fn !== "function") throw new TypeError("fn must be a function");
94	    if (this.#running >= this.#concurrency && this.#queue.isFull())
95	      throw new Error("queue is full");
96	
97	    return new Promise((resolve, reject) => {
98	      const task = { fn, resolve, reject };
99	      if (this.#running < this.#concurrency) {
100	        this.#run(task);
101	      } else {
102	        this.#queue.enqueue(task);
103	      }
104	    });
105	  }
106	
107	  /**
108	   * Wait until all submitted tasks have finished.
109	   */
110	  drain() {
111	    if (this.#running === 0 && this.#queue.isEmpty()) return Promise.resolve();
112	    return new Promise((resolve) => {
113	      this.#drainResolve = resolve;
114	    });
115	  }
116	
117	  /**
118	   * Reject queued tasks, wait for in-flight to finish.
119	   */
120	  async close() {
121	    this.#closed = true;
122	    while (!this.#queue.isEmpty()) {
123	      const t = this.#queue.dequeue();
124	      t.reject(new Error("pool closed before execution"));
125	    }
126	    await this.drain();
127	  }
128	
129	  get running() {
130	    return this.#running;
131	  }
132	
133	  get queued() {
134	    return this.#queue.size;
135	  }
136	
137	  get concurrency() {
138	    return this.#concurrency;
139	  }
140	
141	  // ── internals ──────────────────────────────────────────────────────
142	
143	  #run(task) {
144	    this.#running++;
145	    Promise.resolve().then(() => task.fn()).then(
146	      (val) => {
147	        task.resolve(val);
148	        this.#finish();
149	      },
150	      (err) => {
151	        task.reject(err);
152	        this.#finish();
153	      }
154	    );
155	  }
156	
157	  #finish() {
158	    this.#running--;
159	    if (!this.#queue.isEmpty()) {
160	      this.#run(this.#queue.dequeue());
161	    } else if (this.#running === 0 && this.#drainResolve) {
162	      this.#drainResolve();
163	      this.#drainResolve = null;
164	    }
165	  }
166	}
167	
168	module.exports = { CircularBuffer, ThreadPool };
