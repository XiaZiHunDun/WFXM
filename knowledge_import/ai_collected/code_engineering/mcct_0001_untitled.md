1	"""Fenwick Tree (Binary Indexed Tree) for metrics collection and aggregation.
2	
3	Supports counters (monotonically increasing), gauges (point-in-time values),
4	and histograms (value distributions). Provides O(log n) insert, delete, and
5	range query operations.
6	
7	Example::
8	
9	    tree = FenwickTree()
10	    tree.insert("requests", 1.0, "counter")
11	    tree.insert("requests", 3.0, "counter")
12	    tree.insert("requests", 2.0, "counter")
13	    tree.range_query("requests", 0, 2)  # sum of indices [0..2] -> 6.0
14	"""
15	
16	from __future__ import annotations
17	
18	from dataclasses import dataclass, field
19	from typing import Literal
20	
21	MetricType = Literal["counter", "gauge", "histogram"]
22	
23	
24	@dataclass
25	class Metric:
26	    """A single metric sample."""
27	
28	    name: str
29	    value: float
30	    metric_type: MetricType
31	    tags: dict[str, str] = field(default_factory=dict)
32	
33	    def __hash__(self) -> int:
34	        return hash((self.name, tuple(sorted(self.tags.items()))))
35	
36	
37	class FenwickTree:
38	    """A Fenwick Tree that collects and aggregates metrics.
39	
40	    Each named metric series maintains its own binary-indexed tree over
41	    appended float values.  Operations are O(log n) per series.
42	
43	    Example::
44	
45	        tree = FenwickTree()
46	        tree.insert("latency", 42.5, "histogram", {"host": "a"})
47	        tree.insert("latency", 38.1, "histogram", {"host": "b"})
48	        tree.range_query("latency", 0, 1)   # 80.6
49	        tree.delete("latency", 0)
50	        tree.range_query("latency", 0, 0)   # 38.1
51	    """
52	
53	    def __init__(self) -> None:
54	        self._values: dict[str, list[float]] = {}
55	        self._bit: dict[str, list[float]] = {}
56	        self._size: dict[str, int] = {}
57	        self._metrics: dict[str, Metric] = {}
58	
59	    # -- internal helpers --------------------------------------------------
60	
61	    def _ensure_series(self, name: str) -> None:
62	        """Create bookkeeping structures for *name* if they don't exist."""
63	        if name not in self._values:
64	            self._values[name] = []
65	            self._bit[name] = []
66	            self._size[name] = 0
67	
68	    @staticmethod
69	    def _lsb(i: int) -> int:
70	        """Return the least-significant bit of *i*."""
71	        return i & (-i)
72	
73	    def _bit_update(self, name: str, idx: int, delta: float) -> None:
74	        """Add *delta* at 1-based position *idx* in the BIT for *name*."""
75	        bit = self._bit[name]
76	        cap = len(bit) - 1
77	        while idx <= cap:
78	            bit[idx] += delta
79	            idx += self._lsb(idx)
80	
81	    def _bit_prefix(self, name: str, idx: int) -> float:
82	        """Return the prefix sum ``BIT[1..idx]`` (1-based, inclusive)."""
83	        total = 0.0
84	        bit = self._bit[name]
85	        while idx > 0:
86	            total += bit[idx]
87	            idx -= self._lsb(idx)
88	        return total
89	
90	    # -- public API --------------------------------------------------------
91	
92	    def insert(
93	        self,
94	        name: str,
95	        value: float,
96	        metric_type: MetricType = "gauge",
97	        tags: dict[str, str] | None = None,
98	    ) -> int:
99	        """Append a metric value to its series and update the tree.
100	
101	        Args:
102	            name: Metric name (acts as the series key).
103	            value: Numeric value to record.
104	            metric_type: One of ``"counter"``, ``"gauge"``, or ``"histogram"``.
105	            tags: Optional key/value tags for dimensional filtering.
106	
107	        Returns:
108	            The 0-based index of the newly inserted value within the series.
109	        """
110	        self._ensure_series(name)
111	        tags = tags or {}
112	
113	        values = self._values[name]
114	        self._metrics[name] = Metric(
115	            name=name, value=value, metric_type=metric_type, tags=tags
116	        )
117	
118	        self._size[name] += 1
119	        n = self._size[name]
120	
121	        # Grow the BIT list to accommodate the new 1-based index.
122	        while len(self._bit[name]) <= n:
123	            self._bit[name].append(0.0)
124	
125	        values.append(value)
126	        self._bit_update(name, n, value)
127	
128	        return n - 1  # 0-based index
129	
130	    def delete(self, name: str, index: int) -> float:
131	        """Zero out the value at *index* (soft delete).
132	
133	        The tree structure is kept intact so other indices remain valid;
134	        the deleted slot contributes zero to all future queries.
135	
136	        Args:
137	            name: Metric name.
138	            index: 0-based index to delete.
139	
140	        Returns:
141	            The value that was stored at *index*.
142	
143	        Raises:
144	            KeyError: If *name* is unknown.
145	            IndexError: If *index* is out of range.
146	        """
147	        if name not in self._values:
148	            raise KeyError(f"Unknown metric series: {name!r}")
149	        values = self._values[name]
150	        if index < 0 or index >= len(values):
151	            raise IndexError(f"Index {index} out of range for metric {name!r}")
152	
153	        old = values[index]
154	        if old == 0.0:
155	            return 0.0
156	
157	        values[index] = 0.0
158	        self._bit_update(name, index + 1, -old)
159	        return old
160	
161	    def range_query(self, name: str, left: int, right: int) -> float:
162	        """Return the sum of values in the inclusive range ``[left, right]``.
163	
164	        Args:
165	            name: Metric name.
166	            left: Start index (inclusive, 0-based).
167	            right: End index (inclusive, 0-based).
168	
169	        Returns:
170	            Sum of values in ``[left, right]``.
171	
172	        Raises:
173	            KeyError: If *name* is unknown.
174	            ValueError: If the range is invalid.
175	        """
176	        if name not in self._values:
177	            raise KeyError(f"Unknown metric series: {name!r}")
178	        if left < 0 or right >= len(self._values[name]) or left > right:
179	            raise ValueError(
180	                f"Invalid range [{left}, {right}] for metric {name!r}"
181	            )
182	
183	        return self._bit_prefix(name, right + 1) - self._bit_prefix(name, left)
184	
185	    def prefix_sum(self, name: str, index: int) -> float:
186	        """Return the cumulative sum from index 0 through *index* (inclusive).
187	
188	        Args:
189	            name: Metric name.
190	            index: End index (inclusive, 0-based).
191	
192	        Returns:
193	            Cumulative sum from 0 to *index*.
194	
195	        Raises:
196	            KeyError: If *name* is unknown.
197	            IndexError: If *index* is out of range.
198	        """
199	        if name not in self._values:
200	            raise KeyError(f"Unknown metric series: {name!r}")
201	        if index < 0 or index >= len(self._values[name]):
202	            raise IndexError(
203	                f"Index {index} out of range for metric {name!r}"
204	            )
205	
206	        return self._bit_prefix(name, index + 1)
207	
208	    def count(self, name: str) -> int:
209	        """Return the number of entries in the series (including soft-deleted)."""
210	        return self._size.get(name, 0)
211	
212	    def metric_info(self, name: str) -> Metric:
213	        """Return the most recently recorded :class:`Metric` metadata for *name*."""
214	        if name not in self._metrics:
215	            raise KeyError(f"Unknown metric series: {name!r}")
216	        return self._metrics[name]
217	
218	    def __len__(self) -> int:
219	        return sum(self._size.values())
220	
221	    def __contains__(self, name: str) -> bool:
222	        return name in self._values
223	
224	    def __repr__(self) -> str:
225	        series = ", ".join(f"{k!r}({v})" for k, v in self._size.items())
226	        return f"FenwickTree([{series}])"

**`FenwickTree`** — a clean, standard Binary Indexed Tree with these operations:

| Operation | Method | Time |
|---|---|---|
| Build from array | `FenwickTree([1,3,5,7])` | O(n) |
| Empty tree of size n | `FenwickTree(10)` | O(n) |
| Point update (add delta) | `update(i, delta)` | O(log n) |
| Set value | `set(i, val)` / `ft[i] = val` | O(log n) |
| Prefix sum [0..i] | `prefix_sum(i)` | O(log n) |
| Range sum [l..r] | `range_sum(l, r)` | O(log n) |
| Point query | `point_query(i)` / `ft[i]` | O(log n) |
| Lower bound | `lower_bound(target)` | O(log n) |

`lower_bound` finds the smallest index `i` where `prefix_sum(i) >= target` (requires non-negative values). Supports `len()`, `in`, and `repr` as well.
