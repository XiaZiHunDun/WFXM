1	"""
2	Tarjan's Strongly Connected Components Algorithm
3	
4	Finds all strongly connected components in a directed graph in O(V + E) time
5	using a single DFS pass. An SCC is a maximal set of vertices where every vertex
6	is reachable from every other vertex in the set.
7	
8	Uses an iterative DFS to avoid Python's recursion limit on large graphs.
9	
10	Time:  O(V + E)
11	Space: O(V)
12	"""
13	
14	from collections import defaultdict
15	
16	# Sentinel for the call stack: (v, neighbor_index=0) means "start processing v"
17	# After all neighbors are processed, the lowlink check runs.
18	
19	
20	def tarjan_scc(n, edges):
21	    """Find all SCCs in a directed graph with n vertices (0..n-1).
22	
23	    Args:
24	        n: Number of vertices labeled 0..n-1.
25	        edges: Iterable of (u, v) directed edges.
26	
27	    Returns:
28	        List of SCCs, each SCC is a list of vertex ids.
29	        SCCs are returned in reverse topological order of the component DAG.
30	    """
31	    graph = defaultdict(list)
32	    for u, v in edges:
33	        graph[u].append(v)
34	
35	    index = 0
36	    indices = {}
37	    lowlink = {}
38	    stack = []
39	    on_stack = set()
40	    sccs = []
41	
42	    for start in range(n):
43	        if start in indices:
44	            continue
45	
46	        # Iterative DFS using an explicit call stack
47	        # Each entry: (vertex, next_neighbor_index)
48	        call_stack = [(start, 0)]
49	        indices[start] = lowlink[start] = index
50	        index += 1
51	        stack.append(start)
52	        on_stack.add(start)
53	
54	        while call_stack:
55	            v, ni = call_stack[-1]
56	            neighbors = graph[v]
57	
58	            if ni < len(neighbors):
59	                w = neighbors[ni]
60	                call_stack[-1] = (v, ni + 1)
61	
62	                if w not in indices:
63	                    indices[w] = lowlink[w] = index
64	                    index += 1
65	                    stack.append(w)
66	                    on_stack.add(w)
67	                    call_stack.append((w, 0))
68	                elif w in on_stack:
69	                    lowlink[v] = min(lowlink[v], indices[w])
70	            else:
71	                # All neighbors processed — finalize v
72	                call_stack.pop()
73	
74	                if lowlink[v] == indices[v]:
75	                    scc = []
76	                    while True:
77	                        w = stack.pop()
78	                        on_stack.discard(w)
79	                        scc.append(w)
80	                        if w == v:
81	                            break
82	                    sccs.append(scc)
83	
84	                # Propagate lowlink to parent in DFS tree
85	                if call_stack:
86	                    parent = call_stack[-1][0]
87	                    lowlink[parent] = min(lowlink[parent], lowlink[v])
88	
89	    return sccs
